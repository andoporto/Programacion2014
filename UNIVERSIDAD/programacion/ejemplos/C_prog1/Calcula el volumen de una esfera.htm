Calcula el volumen de una esfera
{-Victor De la Rocha}
{-Algoritmia@groups.msn.com}
{-www.myalgorithm.com}
uses
 crt;
var
 radio,y,v:real;

function esfera(r:real):real;
 begin;
  esfera:=(4/3)*3.1416*(r*r*r);
 end;

begin
 clrscr;
 Write('Radio: ');Readln(radio);
 writeln('Volumen: ',esfera(radio):0:2);
 readkey;
end.
